bước 2:
vào file ts của CustomizeModalComponent

import { Component, inject } from '@angular/core';
import { DialogRef } from '@angular/cdk/dialog';

@Component({
  selector: 'app-customize-modal',
  imports: [],
  templateUrl: './customize-modal.component.html',
  styleUrl: './customize-modal.component.scss'
})
export class CustomizeModalComponent {
  private dialogRef = inject(DialogRef, { optional: true });

  protected closeModal() {
    this.dialogRef?.close();
  }
}


xong rồi ra file .html của CustomizeModalComponent
<p>hệ thống đang quét local storage</p>
<button (click)="closeModal()">Đóng</button>
